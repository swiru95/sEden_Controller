141 void SimpleController::handleMessage(cMessage *msg) {
142 
143  if (msg->isSelfMessage()) {
144  handleSelfMessage(msg);
145  } else if (msg->getKind() == TCP_I_PEER_CLOSED) {
146  int connId =
147  check_and_cast<Indication *>(msg)->getTag<SocketInd>()->getSocketId();
148  delete msg;
149  if (timersMap[connId]->isScheduled()) {
150  cancelAndDelete(timersMap[connId]);
151  }
152  auto request = new Request("close", TCP_C_CLOSE);
153  request->addTagIfAbsent<SocketReq>()->setSocketId(connId);
154  sendBack(request);
155  } else if (msg->getKind() == TCP_I_DATA
156  || msg->getKind() == TCP_I_URGENT_DATA) {
157 
158  Packet *packet = check_and_cast<Packet *>(msg);
159  int connId = packet->getTag<SocketInd>()->getSocketId();
160  ChunkQueue &queue = socketQueue[connId];
161  auto chunk = packet->peekDataAt(B(0), packet->getTotalLength());
162  queue.push(chunk);
163  //HEADER
164  auto& appmsg = queue.pop<Ofp_header>(b(-1), Chunk::PF_ALLOW_NULLPTR);
165  auto version = appmsg->getVersion();
166  auto type = appmsg->getType();
167  if (version == OFPV_v100 or version == OFPV_v110 or version == OFPV_v120
168  or version == OFPV_v130 or version == OFPV_v140
169  or version == OFPV_v150) {
170  auto headLen = appmsg->getLength();
171  auto xid = appmsg->getXid();
172  if (state == CS_ESTABLISHED) {
173  if (timersMap[connId]->isScheduled()) {
174  cancelEvent(timersMap[connId]);
175  timersMap[connId]->setType(ECHO_INTERVAL_TIMER);
176  scheduleAt(simTime() + echoInterval, timersMap[connId]);
177  }
178  }
179  if (type == OFPT_HELLO) {
180  helloRec++;
181  sendHello(connId);
182  sendFeatureReq(connId);
183  if (timersMap[connId]->isScheduled()) {
184  cancelEvent(timersMap[connId]);
185  timersMap[connId]->setType(FEATURE_WAIT_TIMER);
186  scheduleAt(simTime() + featuresWaitInterval,
187  timersMap[connId]);
188  }
189  state = CS_FEATURE_WAIT;
190  } else if (type == OFPT_ECHO_REPLY) {
191  echoRepRec++;
192  } else if (type == OFPT_ECHO_REQUEST and state == CS_ESTABLISHED) {
193  echoReqRec++;
194  sendEchoRep(connId, xid);
195  } else if (type == OFPT_ERROR) {
196  errorRec++;
197  handleOfpError(queue, connId);
198  } else if (type == OFPT_FEATURES_REPLY
199  and (state == CS_FEATURE_WAIT or state == CS_ESTABLISHED)) { //32B
200  state = CS_ESTABLISHED;
201  featRec++;
202  handleOfpFeature(queue, connId, headLen);
203  if (timersMap[connId]->isScheduled()) {
204  cancelEvent(timersMap[connId]);
205  timersMap[connId]->setType(ECHO_INTERVAL_TIMER);
206  scheduleAt(simTime() + echoInterval, timersMap[connId]);
207  }
208  } else if (type == OFPT_GET_CONFIG_REPLY) {
209  getConRepRec++;
210  handleOfpConfig(queue, connId);
211  } else if (type == OFPT_PACKET_IN and state == CS_ESTABLISHED) {
212  packInRec++;
213  handleOfpPacketIn(queue, connId);
214  } else if (type == OFPT_FLOW_REMOVED) {
215  flRemRec++;
216  handleOfpFlowRemoved(queue, connId);
217  } else if (type == OFPT_PORT_STATUS) {
218  portStRec++;
219  handleOfpPortStatus(queue, connId);
220  } else if (type == OFPT_STATS_REPLY) {
221  statRepRec++;
222  handleOfpStat(queue, connId);
223  } else if (type == OFPT_QUEUE_GET_CONFIG_REPLY) {
224  }
225  } else {
226  //TODO Security Risk (if it will be other fake OFP msg (version doestn exists)
227  std::string st =
228  "NOT OFLOW MSG!! SimpleCOntroller Ln115! ConnectionId: ";
229  st.append(std::to_string(connId));
230  perror(st.c_str());
231  }
232  queue.clear();
233  delete msg;
234  } else if (msg->getKind() == TCP_I_AVAILABLE) {
235  socket.processMessage(msg);
236  } else if (msg->getKind() == TCP_I_ESTABLISHED) {
237  EV << "ESTABLISHED!!!" << endl;
238  state = CS_HELLO_WAIT;
239  int connId =
240  check_and_cast<Indication *>(msg)->getTag<SocketInd>()->getSocketId();
241  EV << "1 CONNECTION AT: " << connId << endl;
242  timersMap[connId] = new timer();
243  timersMap[connId]->setConnId(connId);
244  timersMap[connId]->setType(HELLO_WAIT_TIMER);
245  scheduleAt(simTime() + helloWaitInterval, timersMap[connId]);
246  } else {
247  // some indication -- ignore
248  EV_WARN << "drop msg: " << msg->getName() << ", kind:" << msg->getKind()
249  << "("
250  << cEnum::get("inet::TcpStatusInd")->getStringFor(
251  msg->getKind()) << ")\n";
252  delete msg;
253  }

360  auto numPorts = (headLen - 32) / 48;

362  //switch
363  int i = 0;
364  char sname[16], pname[16];
365  while (i < 16) {
366  sname[i] = appPhyPort->getName(i);
367  i++;
368  }
369  //todo if switch exists already!?:D
370  switchMap[sname] = new SwitchProfile(appFeature->getDatapath_id(),
371  appPhyPort->getHw_addr(), appPhyPort->getPort_no(), numPorts,
372  connId);
373  //todo if conn is already!?
374  con2switch[connId] = sname;
375  i = 0;
376  int j = 0;
377  while (i < numPorts - 1) {
378  auto& appPhyPort = queue.pop<Ofp_phy_port>(b(-1),
379  Chunk::PF_ALLOW_NULLPTR);

381  pname[j] = appPhyPort->getName(j);

383  }
384  switchMap[sname]->setEthPorts(pname);
385  switchMap[sname]->setPortHw(appPhyPort->getPort_no(),
386  appPhyPort->getHw_addr());
387  i++;
388  }

447 void SimpleController::handleSelfMessage(cMessage *msg) {
448  timer* t = dynamic_cast<timer*>(msg);
449  if (t) {
450  auto type = t->getType();
451  auto connId = t->getConnId();
452  switch (type) {
453  case ECHO_INTERVAL_TIMER: {
454  sendEchoReq(connId);
455  timersMap[connId]->setType(ECHO_CANCEL_TIMER);
456  scheduleAt(simTime() + echoCancelInterval, timersMap[connId]);
457  return;
458  }
459  case ECHO_CANCEL_TIMER: {
460  auto request = new Request("close", TCP_C_CLOSE);
461  request->addTagIfAbsent<SocketReq>()->setSocketId(connId);
462  sendBack(request);
463  itC2S = con2switch.find(connId);
464  itSM = switchMap.find(itC2S->second);
465  delete itSM->second;
466  itSM->second = nullptr;
467  switchMap.erase(itSM);
468  con2switch.erase(itC2S);
469  itC2S = con2switch.begin();
470  itSM = switchMap.begin();
471  delete msg;
472  msg = nullptr;
473  return;
474  }
475  case PORT_BLOCK_TIMER:
476  //TODO Temporary blocking in apps or sth:D
477  return;
478  case FEATURE_WAIT_TIMER: {
479  auto request = new Request("close", TCP_C_CLOSE);
480  request->addTagIfAbsent<SocketReq>()->setSocketId(connId);
481  sendBack(request);
482  return;
483  }
484  case HELLO_WAIT_TIMER: {
485  auto request = new Request("close", TCP_C_CLOSE);
486  request->addTagIfAbsent<SocketReq>()->setSocketId(connId);
487  sendBack(request);
488  return;
489  }
490  }
491  timersMap.erase(connId);
492  } else {
493  perror("BAD TIMER!");
494  delete msg;
495  msg = nullptr;
496  }
